=== test.rb ===

#!/usr/bin/env ruby

Dir.glob("*.mp4").each do |f|

  File.rename(f, new_f)
end

you can try this

i=1
cd *test*
for name in `ls | sort`;
do
temp=`echo $name | cut -c 1-3`;
if [ $i -lt 10 ]
then
newname=$temp;
newname+="0$i.mp4";
echo $name renamed to $newname;
mv $name $newname
else
newname="$temp$i.mp4";
echo $name renamed to $newname;
mv $name $newname
fi
i=$((i+1));
done
cd ..
